#Back end
npm i express dotenv
npm i -D nodemon
npm i mongoose
npm i cookie-parser
npm i jsonwebtoken bcryptjs
npm install mysql
npm install cors --save
npm i express-mongo-sanitize (No SQL security)
npm i helmet (Security Header)
npm i express-xss-sanitizer (Cross-site scripting security)
npm i express-rate-limit (Rate Limiting)
npm i hpp (Prevent http param pollution)
npm i cors (Cross-Origin Resource Sharing)
npm run dev (run)

npx create-react-app vacqfrontend --template redux (create)
npm install react-router-dom react-toastify axios react-icons
